#include <iostream>
#include <cmath>

using namespace std;

int main()
{
    string a; //Object
    float b = 0.0f; //Ob weight
    string c; //Animal
    float d = 0.0f; //A weight
    float e = 0.0f; //A number
    float t = 0.0f; //time
    float P1 = 0.0f; //Ob power
    
    cout << "Object: ";
    cin >> a;
    cout << a << " weight in kg: ";
    cin >> b;
    cout << "Animal: ";
    cin >> c;
    cout << c << " weight in kg: ";
    cin >> d;
    cout << "Time of motion in seconds: ";
    cin >> t;
    
    b = b * 10;
    d = d * 9.8;
    P1 = b / t; 
    e = t * P1 / d; 
    cout << "Number of " << c << "s necessary to pull the " << a << " for " << t / 60 << "min is " << e << ".";
    
    
     return 0;
}
